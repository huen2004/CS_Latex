A continuación, se presentan las siglas y abreviaturas utilizadas en este documento, junto con su significado completo para facilitar la comprensión.
\begin{description}
	\item[CMMI] \textit{Capability Maturity Model Integration}
	\item[DAR] \textit{Decision Analysis and Resolution}
	\item[GRID] Grupo de Investigación en Redes, Información y Distribución
	\item[SGSI] Sistema de Gestión de Seguridad de la Información
	\item[ISO]  International Organization for Standardization
	\item[UTM] Gestión unificada de amenazas, \textit{Unified Threat Management}
	\item[SMS] Systematic Mapping Study
	\item[IAM] Gestión de identidad y acceso, \textit{Identity and Access Management}
	\item[NPO] Necesidades, Problemas y Oportunidades
	\item[PMV] Producto Minimo Viable
	\item[SIEM] Gestión de información y eventos de seguridad \textit{Security Information and Event Management}
	\item[IDS] Sistema de detección de intrusiones \textit{Intrusion Detection System}
	\item[IPS] Sistema de prevención de intrusiones \textit{Intrusion Prevention System}
	\item[SCI] Science Citation Index
	\item[CVI] Core Value Index
	\item[IRRQ] Index Relation Research Question
	\item[CIA] Confidencialidad, Integridad y Disponibilidad
	\item[VPN] Red privada virtual \textit{Virtual Private Network}
	\item[UML] Lenguaje Unificado de Modelado \textit{Unified Modeling Language}
	\item[DDoS] Denegación de servicio distribuida \textit{Distributed Denial of Service}
\end{description}
